def initialize params_str
  @parameters = {}
  @names = []

  params_str.split("\n").each do |l|
    k,v = l.split("=")
    next unless k && v
    @parameters[k] = v
    @names << k
  end
end

def method_missing(m,args)
  if @parameters.has_key?(m.to_s)
    @parameters[m.to_s]
  else
    super
  end
end
